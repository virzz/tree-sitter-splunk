index=security sourcetype=linux_secure earliest=-24h@h
("Failed password" OR "authentication failure") NOT host=dev*
| rex field=_raw "user[=: ]+(?<user>[^ ]+)"
| eval normalized_user=lower(coalesce(user, "unknown"))
| stats count AS failures, dc(src_ip) AS source_count BY normalized_user, host
| where failures >= 5 AND source_count > 1
| sort - failures
